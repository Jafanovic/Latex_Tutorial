% Line through two points with offset in TikZ
% -------------------------------------------
% https://tex.stackexchange.com/questions/48381/line-through-two-points-with-offset-in-tikz

\documentclass[12pt]{scrartcl}
\usepackage{tikz}
\usetikzlibrary{calc}
\begin{document}

\tikzset{%
    add/.style args={#1 and #2}{
        to path={%
 ($(\tikztostart)!-#1!(\tikztotarget)$)--($(\tikztotarget)!-#2!(\tikztostart)$)%
  \tikztonodes},add/.default={.2 and .2}}
}  

\begin{tikzpicture}
\coordinate (A) at (0,0);
\coordinate (B) at (4,2);
\node at (A) {$\bullet$};
\node at (B) {$\bullet$};
\draw [add= 1 and .5, red, ultra thick] (A) to (B);
\end{tikzpicture}   

\end{document}
